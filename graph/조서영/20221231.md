const validPath = function (n, edges, source, destination) {
    const graph = Array.from({ length: n }, () => Array(n).fill(0));
    edges.forEach(([u, v]) => {
        graph[u][v] = 1;
        graph[v][u] = 1;
    });
    let ch = Array(n).fill(0);
    ch[source] = 1;
    let result = false;
    function DFS(L) {
        if (L === destination) {
            result = true;
            return;
        } else {
            graph[L].forEach((ele, i) => {
                if (ele === 1 && ch[i] === 0) {
                    ch[i] = 1;
                    DFS(i);
                    ch[i] = 0;
                }
            });
        }
    }
    DFS(source);

    return result;
};

const validPath = function (n, edges, source, destination) {
    const graph = Array.from(Array(n), () => Array(n).fill(0));
    edges.forEach(([u, v]) => {
        graph[u][v] = 1;
        graph[v][u] = 1;
    });
    let result = false;
    let ch = Array(n).fill(0);
    ch[source] = 1;
    let queue = [source];
    while (queue.length) {
        let v = queue.shift();
        if (v === destination) {
            result = true;
            break;
        }
        for (let i = 0; i < n; i++) {
            if (graph[v][i] === 1 && ch[i] === 0) {
                ch[i] = 1;
                queue.push(i);
            }
        }
    }

    return result;
};

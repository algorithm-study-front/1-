/**
 * @param {number} n
 * @param {number[][]} edges
 * @param {number} source
 * @param {number} destination
 * @return {boolean}
 */

let answer = 0;

const dfs = (v, target, adj, count, n) => {
  if (count > n) {
    return;
  }

  if (v === target) {
    answer++;
  }

  for (const e of adj[v]) {
    dfs(e, target, adj, count + 1, n);
  }
};

var validPath = function (n, edges, source, destination) {
  const adj = Array.from({ length: n }, () => []);

  edges.forEach(([u, v]) => {
    adj[u].push(v);
    adj[v].push(u);
  });

  dfs(source, destination, adj, 0, n);
  console.log("answer", answer);
  return answer;
};

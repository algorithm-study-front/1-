## [Inflearn 강의] 순열 구하기

```js
function solution(m, arr) {
    const { length } = arr;
    const memo = Array(m);
    const visited = Array(length).fill(0);

    function dfs(depth) {
        if (depth === m) {
            console.log(memo.join(' '));
        } else {
            for (let i = 0; i < length; i++) {
                if (visited[i] === 0) {
                    visited[i] = 1;
                    memo[depth] = arr[i];
                    dfs(depth + 1);
                    visited[i] = 0;
                }
            }
        }
    }
    dfs(0);
}
```

## [Inflearn 강의] 팩토리얼

```js
function solution(n) {
    if (n === 1) {
        return 1;
    }
    return n * solution(n - 1);
}
```

## [Inflearn 강의] 조합의 경우수(메모이제이션)

-   강의 내용 참조

```js
function solution(n, r) {
    function dfs(n, r) {
        if (n === r || r === 0) {
            return 1;
        } else {
            return dfs(n - 1, r - 1) + dfs(n - 1, r);
        }
    }
    return dfs(n, r);
}
```

-   메모이제이션 기능 추가

```js
function solution(n, r) {
    const memo = Array.from(Array(n + 1), () => Array(r + 1).fill(0));
    function dfs(n, r) {
        if (memo[n][r] > 0) {
            return memo[n][r];
        }
        if (n === r || r === 0) {
            return 1;
        } else {
            return (memo[n][r] = dfs(n - 1, r - 1) + dfs(n - 1, r));
        }
    }
    return dfs(n, r);
}
```

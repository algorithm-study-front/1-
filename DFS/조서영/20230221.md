## [Inflearn 강의] 경로탐색(인접행렬)

```js
function solution(n, arr) {
    let answer = 0;
    let matrix = Array.from(Array(n + 1), () => Array(n + 1).fill(0));
    let visited = Array(n + 1).fill(0);

    for (let [a, b] of arr) {
        matrix[a][b] = 1;
    }

    function dfs(v) {
        if (v === n) {
            answer++;
        } else {
            for (let i = 1; i <= n; i++) {
                if (matrix[v][i] && !visited[i]) {
                    visited[i] = 1;
                    dfs(i);
                    visited[i] = 0;
                }
            }
        }
    }
    visited[1] = 1;
    dfs(1);

    return answer;
}
```

## [Inflearn 강의] 경로탐색(인접리스트)

```js
function solution(n, arr) {
    let answer = 0;
    let matrix = Array.from(Array(n + 1), () => Array());
    let visited = Array(n + 1).fill(0);

    arr.forEach(([a, b]) => {
        matrix[a].push(b);
    });

    function dfs(v) {
        if (v === n) answer++;
        else {
            // FEEDBACK: for .. of로 변환
            for (let i = 0; i < matrix[v].length; i++) {
                let idx = matrix[v][i];
                if (visited[idx] === 0) {
                    visited[idx] = 1;
                    dfs(idx);
                    visited[idx] = 0;
                }
            }
        }
    }

    visited[1] = 1;
    dfs(1);
    return answer;
}
```

## [Inflearn 강의] 미로탐색

```js
function solution(board) {
    let answer = 0;
    const dx = [-1, 0, 1, 0];
    const dy = [0, 1, 0, -1];

    function dfs(x, y) {
        if (x === 6 && y === 6) answer++;
        else {
            for (let k = 0; k < 4; k++) {
                const nx = x + dx[k];
                const ny = y + dy[k];
                // FEEDBACK: 조건문 변수화 하기
                if (
                    nx >= 0 &&
                    nx <= 6 &&
                    ny >= 0 &&
                    ny <= 6 &&
                    board[nx][ny] === 0
                ) {
                    board[nx][ny] = 1;
                    dfs(nx, ny);
                    board[nx][ny] = 0;
                }
            }
        }
    }
    board[0][0] = 1;
    dfs(0, 0);

    return answer;
}
```

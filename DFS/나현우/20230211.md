# 조합 구하기

```js
const solution = (total, count) => {
  let answer = [];
  let tmp = Array.from({ length: count }, () => 0);
  const dfs = (level, start) => {
    if (level === count) {
      answer.push([...tmp]);
      return;
    } else {
      for (let i = start; i <= total; i++) {
        tmp[level] = i;
        dfs(level + 1, i + 1);
      }
    }
  };
  dfs(0, 1);
  return answer;
};

console.log(solution(4, 2));
```

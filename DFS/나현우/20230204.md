# 순열 구하기

```js
const solution = (count, arr) => {
  let answer = 0;

  const dfs = (level, str) => {
    if (level === count) {
      console.log(str);
      answer += 1;
      return;
    } else {
      // 위에서 선택한걸 어떻게 제거하지 ?
      arr.forEach((char) => {
        dfs(level + 1, str + " " + char);
      });
    }
  };

  dfs(0, "");

  return answer;
};

let arr = [3, 6, 9];
console.log(solution(2, arr));

```

# 팩토리얼

```js
const solution = (n) => {
  let answer = 1;
  const dfs = (n) => {
    if (n === 1) {
      return;
    } else {
      answer *= n;
      dfs(n - 1);
    }
  };

  dfs(n);

  return answer;
};

console.log(solution(5));


```

# 조합의 경우수 (메모이제이션)

```js
const solution = (n, r) => {
  let answer = 0;

  const dfs = (n, r) => {
    if (n === 2) {
      answer += 3;
      return;
    } else {
      dfs(n - 1, r);
      dfs(n - 1, r - 1);
    }
  };

  dfs(n, r);

  return answer;
};

console.log(solution(5, 3));

```

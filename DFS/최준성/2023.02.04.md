## 순열 구하기

```js
const solution = (n, m, numbers) => {
  const answer = [];
  let temp = Array.from({ length: m }, () => 0);

  const dfs = (depth) => {
    if (depth === m) {
      answer.push([...temp]);
    } else {
      for (let i = 0; i < n; i++) {
        // FEEDBACK: 이전의 모든 값을 검증할 필요가 있어보입니다.
        if (temp[depth - 1] === numbers[i]) {
          continue;
        }
        temp[depth] = numbers[i];
        dfs(depth + 1);
      }
    }
  };

  dfs(0);

  return answer;
};

const numbers = [3, 6, 9];
const result = solution(3, 2, numbers);
console.log(solution(3, 2, numbers), result.length);
```

## 팩토리얼

<!-- FEEDBACK: dfs 인자 한개로. -->
```js
const solution = (n) => {
  const dfs = (num, acc) => {
    if (num === 1) {
      return acc;
    }

    return dfs(num - 1, (num - 1) * acc);
  };

  return dfs(n, n);
};

console.log(solution(5));
```

## 조합의 경우수(메모이제이션)

```js
const solution = (num, m) => {
  const dfs = (n, r) => {
    if (n === r || r === 0) {
      return 1;
    }

    return dfs(n - 1, r - 1) + dfs(n - 1, r);
  };

  return dfs(num, m);
};

console.log(solution(5, 3));
```

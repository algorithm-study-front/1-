# DFS 수열 추측하기

```js
const solution = (n, r) => {
  let answer = [];
  let checked = Array.from({ length: n + 1 }, () => false);

  const dfs = (level, sum) => {
    if (level === n && sum === r) {
      return;
    }

    for (let i = 1; i <= n; i++) {
      if (!checked[i]) {
        checked[i] = true;
        // dfs(level + 1, ....);
        checked[i] = false;
      }
    }
  };

  dfs(0, 0);
  return answer;
};

solution(4, 16);
```

# DFS 조합 구하기

```js
const solution = (n, m) => {
  const answer = [];
  const temp = Array.from({ length: m }, () => 0);

  const dfs = (level, start) => {
    if (level === m) {
      answer.push([...temp]);
      return;
    }

    for (let i = start; i <= n; i++) {
      temp[level] = i;
      dfs(level + 1, i + 1);
    }
  };

  dfs(0, 1);

  return [...answer, answer.length];
};

solution(4, 2);
```

# DFS 수들의 조합

```js
const solution = (k, arr, m) => {
  let answer = 0;
  const len = arr.length;
  const temp = Array.from({ length: k });

  const dfs = (level, s, sum) => {
    if (level === k && sum % m === 0) {
      answer++;
      return;
    }

    for (let i = s; i < len; i++) {
      temp[level] = arr[i];
      dfs(level + 1, i + 1, sum + arr[i]);
    }
  };

  dfs(0, 0, 0);

  return answer;
};

solution(3, [2, 4, 5, 8, 12], 6);
```
